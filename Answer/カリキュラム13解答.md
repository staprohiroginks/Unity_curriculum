`MainWindow.axml.cs`ファイルの内容は以下のようになっています。
```cs
using Avalonia.Controls;

namespace ExamCalc;

public partial class MainWindow : Window
{
    public MainWindow()
    {
        InitializeComponent();
        CalcButton.Click += Calc_Click;
    }

    private void Calc_Click(object? sender, Avalonia.Interactivity.RoutedEventArgs e)
    {
        SumScoreLabel.Content = "合計点:計算できません";
        AveScoreLabel.Content = "平均点:計算できません";
        int JapaneseScore, MathScore,ScienceScore,SocialStudiesScore,EnglishScore;
        if(int.TryParse(JapaneseTextBox.Text,out JapaneseScore)==false) 
        {
            return;
        }
        if(int.TryParse(MathTextBox.Text,out MathScore)==false) 
        {
            return;
        }
        if(int.TryParse(ScienceTextBox.Text,out ScienceScore)==false) 
        {
            return;
        }
        if(int.TryParse(SocialStudiesTextBox.Text,out SocialStudiesScore)==false) 
        {
            return;
        }
        if(int.TryParse(EnglishTextBox.Text,out EnglishScore)==false) 
        {
            return;
        }
        int SumScore = JapaneseScore + MathScore + ScienceScore + EnglishScore;
        SumScoreLabel.Content = "合計点:" + SumScore.ToString();
        double AveScore = (double)SumScore / 5.0; 
        AveScoreLabel.Content = "平均点:" + AveScore.ToString();
    }
}
```
